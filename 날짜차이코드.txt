		public static long diffOfDate(String a, String b) throws Exception {
				
			
		        SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
		        
		        Date today = format.parse(a);
		        Date regdt = format.parse(b);
		        
		        long diff = today.getTime() - regdt.getTime();
		        long diffDays = diff / (24 * 60 * 60 * 1000);
		        
		        return diffDays;
		    }
